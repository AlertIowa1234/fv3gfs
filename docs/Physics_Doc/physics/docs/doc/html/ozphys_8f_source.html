48 ! as are all other variables

53 !
54  real, parameter :: gravi=1.0/grav
55  integer im, ix, levs, ko3, pl_coeff,me
56  real(kind=kind_phys) ozi(ix,levs), ozo(ix,levs), po3(ko3),
57  & prsl(ix,levs), tin(ix,levs), delp(ix,levs),

59  & ozp(ix,levs,pl_coeff), dt
60 !

64  real(kind=kind_phys) wk1(im), wk2(im), wk3(im), prod(im,pl_coeff),
65  & ozib(im), colo3(im,levs+1)
66 !
67  if (pl_coeff > 2) then
68  colo3(:,levs+1) = 0.0
69  do l=levs,1,-1
70  do i=1,im

72  enddo
73  enddo

77  pmin = 1.0e10

79 !

90  if (pmax < po3(k)) kmin = k
91  enddo

98  flg(i) = .true.
99  wk2(i) = (wk1(i) - po3(k+1)) * temp
100  wk3(i) = 1.0 - wk2(i)
101  endif
102  enddo
103  do j=1,pl_coeff
104  do i=1,im
105  if (flg(i)) then
106  prod(i,j) = wk2(i) * prdout(i,k,j)
107  & + wk3(i) * prdout(i,k+1,j)
108  endif
109  enddo
110  enddo
111  enddo
112 !

114  do i=1,im

117  endif
118  if (wk1(i) >= po3(1)) then

120  endif
121  enddo
122  enddo

133  enddo
134  endif
135  endif

139  tem = prod(i,1) + prod(i,3)*tin(i,l)
140  & + prod(i,4)*colo3(i,l+1)
141 ! if (me .eq. 0) print *,'ozphys tem=',tem,' prod=',prod(i,:)
142 ! &,' ozib=',ozib(i),' l=',l,' tin=',tin(i,l),'colo3=',colo3(i,l+1)
143  ozo(i,l) = (ozib(i) + tem*dt) / (1.0 + prod(i,2)*dt)
144  enddo
145  if (ldiag3d) then ! ozone change diagnostics
146  do i=1,im
147  ozp(i,l,1) = ozp(i,l,1) + prod(i,1)*dt
148  ozp(i,l,2) = ozp(i,l,2) + (ozo(i,l) - ozib(i))
149  ozp(i,l,3) = ozp(i,l,3) + prod(i,3)*tin(i,l)*dt
150  ozp(i,l,4) = ozp(i,l,4) + prod(i,4)*colo3(i,l+1)*dt
151  enddo
152  endif

154  enddo ! vertical loop
155 !
