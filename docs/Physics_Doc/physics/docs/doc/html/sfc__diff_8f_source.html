1  subroutine sfc_diff(im,ps,u1,v1,t1,q1,z1,
2  & snwdph,tskin,z0rl,cm,ch,rb,
3  & prsl1,prslki,islimsk,
4  & stress,fm,fh,
5  & ustar,wind,ddvel,fm10,fh2,
6  & sigmaf,vegtype,shdmax,ivegsrc,
7  & tsurf,flag_iter,redrag)
8 !
9  use machine , only : kind_phys
10  use funcphys, only : fpvs
11  use physcons, grav => con_g, cp => con_cp
12  &, rvrdm1 => con_fvirt, rd => con_rd
13  &, eps => con_eps, epsm1 => con_epsm1
14 
15  implicit none
16 !
17  integer im, ivegsrc
18  real(kind=kind_phys), dimension(im) :: ps, u1, v1, t1, q1, z1
19  &, tskin, z0rl, cm, ch, rb
20  &, prsl1, prslki, stress
21  &, fm, fh, ustar, wind, ddvel
22  &, fm10, fh2, sigmaf, shdmax
23  &, tsurf, snwdph
24  integer, dimension(im) :: vegtype, islimsk
25 
26  logical flag_iter(im) ! added by s.lu
27  logical redrag ! reduced drag coeff. flag for high wind over sea (j.han)
28 !
29 ! locals
30 !
31  integer i
32 !
33  real(kind=kind_phys) aa, aa0, bb, bb0, dtv, adtv,qs1,
34  & hl1, hl12, pm, ph, pm10, ph2, rat,
35  & thv1, tvs, z1i, z0, z0max, ztmax,
36  & fms, fhs, hl0, hl0inf, hlinf,
37  & hl110, hlt, hltinf, olinf,
38  & restar, czilc, tem1, tem2, ztmax1
39 !
40  real(kind=kind_phys), parameter ::
41  & charnock=.014, ca=.4 ! ca - von karman constant
42  &, z0s_max=.317e-2 ! a limiting value at high winds over sea
43 
44  &, alpha=5., a0=-3.975, a1=12.32, alpha4=4.0*alpha
45  &, b1=-7.755, b2=6.041, alpha2=alpha+alpha, beta=1.0
46  &, a0p=-7.941, a1p=24.75, b1p=-8.705, b2p=7.899
47  &, vis=1.4e-5, rnu=1.51e-5, visi=1.0/vis
48 
49  &, log01=log(0.01), log05=log(0.05), log07=log(0.07)
50  &, ztmin1=-999.0
51 
52 ! parameter (charnock=.014,ca=.4)!c ca is the von karman constant
53 ! parameter (alpha=5.,a0=-3.975,a1=12.32,b1=-7.755,b2=6.041)
54 ! parameter (a0p=-7.941,a1p=24.75,b1p=-8.705,b2p=7.899,vis=1.4e-5)
55 
56 ! real(kind=kind_phys) aa1,bb1,bb2,cc,cc1,cc2,arnu
57 ! parameter (aa1=-1.076,bb1=.7045,cc1=-.05808)
58 ! parameter (bb2=-.1954,cc2=.009999)
59 ! parameter (arnu=.135*rnu)
60 !
61 ! z0s_max=.196e-2 for u10_crit=25 m/s
62 ! z0s_max=.317e-2 for u10_crit=30 m/s
63 ! z0s_max=.479e-2 for u10_crit=35 m/s
64 !
65 ! mbek -- toga-coare flux algorithm
66 ! parameter (rnu=1.51e-5,arnu=0.11*rnu)
67 !
68 ! initialize variables. all units are supposedly m.k.s. unless specified
69 ! ps is in pascals, wind is wind speed,
70 ! surface roughness length is converted to m from cm
71 !
72  do i=1,im
73  if(flag_iter(i)) then
74  wind(i) = max(sqrt(u1(i)*u1(i) + v1(i)*v1(i))
75  & + max(0.0, min(ddvel(i), 30.0)), 1.0)
76  tem1 = 1.0 + rvrdm1 * max(q1(i),1.e-8)
77  thv1 = t1(i) * prslki(i) * tem1
78  tvs = 0.5 * (tsurf(i)+tskin(i)) * tem1
79  qs1 = fpvs(t1(i))
80  qs1 = max(1.0e-8, eps * qs1 / (prsl1(i) + epsm1 * qs1))
81 
82  z0 = 0.01 * z0rl(i)
83  z0max = max(1.0e-6, min(z0,z1(i)))
84  z1i = 1.0 / z1(i)
85 
86 ! compute stability dependent exchange coefficients
87 ! this portion of the code is presently suppressed
88 !
89 
90  if(islimsk(i) == 0) then ! over ocean
91  ustar(i) = sqrt(grav * z0 / charnock)
92 
93 !** test xubin's new z0
94 
95 ! ztmax = z0max
96 
97  restar = max(ustar(i)*z0max*visi, 0.000001)
98 
99 ! restar = log(restar)
100 ! restar = min(restar,5.)
101 ! restar = max(restar,-5.)
102 ! rat = aa1 + (bb1 + cc1*restar) * restar
103 ! rat = rat / (1. + (bb2 + cc2*restar) * restar))
104 ! rat taken from zeng, zhao and dickinson 1997
105 
106  rat = min(7.0, 2.67 * sqrt(sqrt(restar)) - 2.57)
107  ztmax = z0max * exp(-rat)
108 
109  else ! over land and sea ice
110 !** xubin's new z0 over land and sea ice
111  tem1 = 1.0 - shdmax(i)
112  tem2 = tem1 * tem1
113  tem1 = 1.0 - tem2
114 
115  if( ivegsrc == 1 ) then
116 
117  if (vegtype(i) == 10) then
118  z0max = exp( tem2*log01 + tem1*log07 )
119  elseif (vegtype(i) == 6) then
120  z0max = exp( tem2*log01 + tem1*log05 )
121  elseif (vegtype(i) == 7) then
122 ! z0max = exp( tem2*log01 + tem1*log01 )
123  z0max = 0.01
124  elseif (vegtype(i) == 16) then
125 ! z0max = exp( tem2*log01 + tem1*log01 )
126  z0max = 0.01
127  else
128  z0max = exp( tem2*log01 + tem1*log(z0max) )
129  endif
130 
131  elseif (ivegsrc == 2 ) then
132 
133  if (vegtype(i) == 7) then
134  z0max = exp( tem2*log01 + tem1*log07 )
135  elseif (vegtype(i) == 8) then
136  z0max = exp( tem2*log01 + tem1*log05 )
137  elseif (vegtype(i) == 9) then
138 ! z0max = exp( tem2*log01 + tem1*log01 )
139  z0max = 0.01
140  elseif (vegtype(i) == 11) then
141 ! z0max = exp( tem2*log01 + tem1*log01 )
142  z0max = 0.01
143  else
144  z0max = exp( tem2*log01 + tem1*log(z0max) )
145  endif
146 
147  endif
148  z0max = max(z0max,1.0e-6)
149 !
150 ! czilc = 10.0 ** (- (0.40/0.07) * z0) ! fei's canopy height dependance of czil
151  czilc = 0.8
152 
153  tem1 = 1.0 - sigmaf(i)
154  ztmax = z0max*exp( - tem1*tem1
155  & * czilc*ca*sqrt(ustar(i)*(0.01/1.5e-05)))
156 
157  endif ! end of if(islimsk(i) == 0) then
158 
159  ztmax = max(ztmax,1.0e-6)
160  tem1 = z0max/z1(i)
161  if (abs(1.0-tem1) > 1.0e-6) then
162  ztmax1 = - beta*log(tem1)/(alpha2*(1.-tem1))
163  else
164  ztmax1 = 99.0
165  endif
166  if( z0max < 0.05 .and. snwdph(i) < 10.0 ) ztmax1 = 99.0
167 
168 
169 ! compute stability indices (rb and hlinf)
170 
171  dtv = thv1 - tvs
172  adtv = max(abs(dtv),0.001)
173  dtv = sign(1.,dtv) * adtv
174  rb(i) = max(-5000.0, (grav+grav) * dtv * z1(i)
175  & / ((thv1 + tvs) * wind(i) * wind(i)))
176  tem1 = 1.0 / z0max
177  tem2 = 1.0 / ztmax
178  fm(i) = log((z0max+z1(i)) * tem1)
179  fh(i) = log((ztmax+z1(i)) * tem2)
180  fm10(i) = log((z0max+10.) * tem1)
181  fh2(i) = log((ztmax+2.) * tem2)
182  hlinf = rb(i) * fm(i) * fm(i) / fh(i)
183  hlinf = min(max(hlinf,ztmin1),ztmax1)
184 !
185 ! stable case
186 !
187  if (dtv >= 0.0) then
188  hl1 = hlinf
189  if(hlinf > .25) then
190  tem1 = hlinf * z1i
191  hl0inf = z0max * tem1
192  hltinf = ztmax * tem1
193  aa = sqrt(1. + alpha4 * hlinf)
194  aa0 = sqrt(1. + alpha4 * hl0inf)
195  bb = aa
196  bb0 = sqrt(1. + alpha4 * hltinf)
197  pm = aa0 - aa + log( (aa + 1.)/(aa0 + 1.) )
198  ph = bb0 - bb + log( (bb + 1.)/(bb0 + 1.) )
199  fms = fm(i) - pm
200  fhs = fh(i) - ph
201  hl1 = fms * fms * rb(i) / fhs
202  hl1 = min(max(hl1, ztmin1), ztmax1)
203  endif
204 !
205 ! second iteration
206 !
207  tem1 = hl1 * z1i
208  hl0 = z0max * tem1
209  hlt = ztmax * tem1
210  aa = sqrt(1. + alpha4 * hl1)
211  aa0 = sqrt(1. + alpha4 * hl0)
212  bb = aa
213  bb0 = sqrt(1. + alpha4 * hlt)
214  pm = aa0 - aa + log( (1.0+aa)/(1.0+aa0) )
215  ph = bb0 - bb + log( (1.0+bb)/(1.0+bb0) )
216  hl110 = hl1 * 10. * z1i
217  hl110 = min(max(hl110, ztmin1), ztmax1)
218  aa = sqrt(1. + alpha4 * hl110)
219  pm10 = aa0 - aa + log( (1.0+aa)/(1.0+aa0) )
220  hl12 = (hl1+hl1) * z1i
221  hl12 = min(max(hl12,ztmin1),ztmax1)
222 ! aa = sqrt(1. + alpha4 * hl12)
223  bb = sqrt(1. + alpha4 * hl12)
224  ph2 = bb0 - bb + log( (1.0+bb)/(1.0+bb0) )
225 !
226 ! unstable case - check for unphysical obukhov length
227 !
228  else ! dtv < 0 case
229  olinf = z1(i) / hlinf
230  tem1 = 50.0 * z0max
231  if(abs(olinf) <= tem1) then
232  hlinf = -z1(i) / tem1
233  hlinf = min(max(hlinf,ztmin1),ztmax1)
234  endif
235 !
236 ! get pm and ph
237 !
238  if (hlinf >= -0.5) then
239  hl1 = hlinf
240  pm = (a0 + a1*hl1) * hl1 / (1.+ (b1+b2*hl1) *hl1)
241  ph = (a0p + a1p*hl1) * hl1 / (1.+ (b1p+b2p*hl1)*hl1)
242  hl110 = hl1 * 10. * z1i
243  hl110 = min(max(hl110, ztmin1), ztmax1)
244  pm10 = (a0 + a1*hl110) * hl110 / (1.+(b1+b2*hl110)*hl110)
245  hl12 = (hl1+hl1) * z1i
246  hl12 = min(max(hl12, ztmin1), ztmax1)
247  ph2 = (a0p + a1p*hl12) * hl12 / (1.+(b1p+b2p*hl12)*hl12)
248  else ! hlinf < 0.05
249  hl1 = -hlinf
250  tem1 = 1.0 / sqrt(hl1)
251  pm = log(hl1) + 2. * sqrt(tem1) - .8776
252  ph = log(hl1) + .5 * tem1 + 1.386
253 ! pm = log(hl1) + 2.0 * hl1 ** (-.25) - .8776
254 ! ph = log(hl1) + 0.5 * hl1 ** (-.5) + 1.386
255  hl110 = hl1 * 10. * z1i
256  hl110 = min(max(hl110, ztmin1), ztmax1)
257  pm10 = log(hl110) + 2.0 / sqrt(sqrt(hl110)) - .8776
258 ! pm10 = log(hl110) + 2. * hl110 ** (-.25) - .8776
259  hl12 = (hl1+hl1) * z1i
260  hl12 = min(max(hl12, ztmin1), ztmax1)
261  ph2 = log(hl12) + 0.5 / sqrt(hl12) + 1.386
262 ! ph2 = log(hl12) + .5 * hl12 ** (-.5) + 1.386
263  endif
264 
265  endif ! end of if (dtv >= 0 ) then loop
266 !
267 ! finish the exchange coefficient computation to provide fm and fh
268 !
269  fm(i) = fm(i) - pm
270  fh(i) = fh(i) - ph
271  fm10(i) = fm10(i) - pm10
272  fh2(i) = fh2(i) - ph2
273  cm(i) = ca * ca / (fm(i) * fm(i))
274  ch(i) = ca * ca / (fm(i) * fh(i))
275  tem1 = 0.00001/z1(i)
276  cm(i) = max(cm(i), tem1)
277  ch(i) = max(ch(i), tem1)
278  stress(i) = cm(i) * wind(i) * wind(i)
279  ustar(i) = sqrt(stress(i))
280 !
281 ! update z0 over ocean
282 !
283  if(islimsk(i) == 0) then
284  z0 = (charnock / grav) * ustar(i) * ustar(i)
285 
286 ! mbek -- toga-coare flux algorithm
287 ! z0 = (charnock / grav) * ustar(i)*ustar(i) + arnu/ustar(i)
288 ! new implementation of z0
289 ! cc = ustar(i) * z0 / rnu
290 ! pp = cc / (1. + cc)
291 ! ff = grav * arnu / (charnock * ustar(i) ** 3)
292 ! z0 = arnu / (ustar(i) * ff ** pp)
293 
294  if (redrag) then
295  z0rl(i) = 100.0 * max(min(z0, z0s_max), 1.e-7)
296  else
297  z0rl(i) = 100.0 * max(min(z0,.1), 1.e-7)
298  endif
299  endif
300  endif ! end of if(flagiter) loop
301  enddo
302 
303  return
304  end
